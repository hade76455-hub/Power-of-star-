            createBattleEffect() {
                // ÿ™ÿ£ÿ´Ÿäÿ± ÿ®ÿ≥Ÿäÿ∑ ŸÅŸä Ÿàÿ≥ÿ∑ ÿßŸÑÿ¥ÿßÿ¥ÿ©
                this.battleEffects = this.battleEffects || [];
                const effect = {
                    x: this.canvas.width / 2,
                    y: this.canvas.height / 2,
                    radius: 50,
                    alpha: 1.0,
                    update: function() { this.alpha -= 0.02; this.radius += 2; },
                    draw: function(ctx) {
                        ctx.globalAlpha = this.alpha;
                        ctx.fillStyle = '#ff6b6b';
                        ctx.beginPath();
                        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                        ctx.fill();
                        ctx.globalAlpha = 1.0;
                    }
                };
                this.battleEffects.push(effect);
            }
